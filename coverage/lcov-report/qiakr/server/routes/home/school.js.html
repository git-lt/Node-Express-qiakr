/**
 * [洽客大学页面路由模块]
 */
 
var videoApi = require('../../api/home/videos');
var loger = require('../../common/logHelper').log;
 
var schoolRouter = {
	school:function(req, res, next){
 
		res.render('home/school', {title:'洽客大学'});
	},
	video:function(req, res, next){
		var pms = {
			inde: req.params.type || '0',
			length: req.body.length || 6,
			fuzzyName: req.body.fuzzyName || '',
			type: req.params.type || 1,
			status: 2
		}
		// 获取数据
		videoApi.getQiakrVideoList(pms).then(function(data){
			// loger.info(data);
			res.render('home/videos', {title:'在线教程', data:data});
		}).catch(function(err){
			loger.error('request failure!');
			next();
		});
	},
	data_video_list:function(req, res, next){
		var pms = {
			index: req.body.index || '0',
			length: req.body.length || 6,
			fuzzyName: req.body.fuzzyName || '',
			type: req.params.type || 1,
			status: 2
		}
		// loger.info(pms);
			videoApi.getQiakrVideoList(pms).then(function(data){
			res.send({status: '0', data: data});
		}).catch(function(err){
			res.send({status: '1', errmsg:'视频列表数据请求失败！'});
		});
	},
	videoDetail:function(req, res, next){
		var id = req.params.id;
		if(id !== undefined){
			var pms ={
				id: id
			}
			videoApi.getQiakrVideoById(pms).then(function(data){
				res.render('home/newsDetail', data);
			}).catch(function(err){
				loger.error('Request failed!');
				next();
			});
		}else{
			next();
		}
	},
	data_video_detail:function(req, res, next){
		var pms = {
			qiakrVideoId: req.params.id || 1
		}
		videoApi.getQiakrVideoById(pms).then(function(data){
			res.send({status: '0', data: data});
		}).catch(function(err){
			res.send({status: '1', errmsg:'视频详情数据请求失败！'})
		});
	}
}
 
module.exports = schoolRouter;
